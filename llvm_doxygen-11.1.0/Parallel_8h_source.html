35 class Latch {
36  uint32_t Count;
37  mutable std::mutex Mutex;
38  mutable std::condition_variable Cond;
39 
40 public:
41  explicit Latch(uint32_t Count = 0) : Count(Count) {}
42  ~Latch() { sync(); }
43 
44  void inc() {
45  std::lock_guard<std::mutex> lock(Mutex);
46  ++Count;
47  }
48 
49  void dec() {
50  std::lock_guard<std::mutex> lock(Mutex);
51  if (--Count == 0)
52  Cond.notify_all();
53  }
54 
55  void sync() const {
56  std::unique_lock<std::mutex> lock(Mutex);
57  Cond.wait(lock, [&] { return Count == 0; });
58  }

76 /// Inclusive median.
77 template <class RandomAccessIterator, class Comparator>
78 RandomAccessIterator medianOf3(RandomAccessIterator Start,
79  RandomAccessIterator End,
80  const Comparator &Comp) {
81  RandomAccessIterator Mid = Start + (std::distance(Start, End) / 2);
82  return Comp(*Start, *(End - 1))
83  ? (Comp(*Mid, *(End - 1)) ? (Comp(*Start, *Mid) ? Mid : Start)
84  : End - 1)
85  : (Comp(*Mid, *Start) ? (Comp(*(End - 1), *Mid) ? Mid : End - 1)
86  : Start);
87 }

89 template <class RandomAccessIterator, class Comparator>
90 void parallel_quick_sort(RandomAccessIterator Start, RandomAccessIterator End,
91  const Comparator &Comp, TaskGroup &TG, size_t Depth) {
92  // Do a sequential sort for small inputs.
93  if (std::distance(Start, End) < detail::MinParallelSize || Depth == 0) {
94  llvm::sort(Start, End, Comp);
95  return;
96  }
97 
98  // Partition.
99  auto Pivot = medianOf3(Start, End, Comp);
100  // Move Pivot to End.
101  std::swap(*(End - 1), *Pivot);
102  Pivot = std::partition(Start, End - 1, [&Comp, End](decltype(*Start) V) {
103  return Comp(V, *(End - 1));
104  });
105  // Move Pivot to middle of partition.
106  std::swap(*Pivot, *(End - 1));
107 
108  // Recurse.
109  TG.spawn([=, &Comp, &TG] {
110  parallel_quick_sort(Start, Pivot, Comp, TG, Depth - 1);
111  });
112  parallel_quick_sort(Pivot + 1, End, Comp, TG, Depth - 1);
113 }

115 template <class RandomAccessIterator, class Comparator>
116 void parallel_sort(RandomAccessIterator Start, RandomAccessIterator End,
117  const Comparator &Comp) {
118  TaskGroup TG;
119  parallel_quick_sort(Start, End, Comp, TG,
120  llvm::Log2_64(std::distance(Start, End)) + 1);
121 }

123 template <class IterTy, class FuncTy>
124 void parallel_for_each(IterTy Begin, IterTy End, FuncTy Fn) {
125  // TaskGroup has a relatively high overhead, so we want to reduce
126  // the number of spawn() calls. We'll create up to 1024 tasks here.
127  // (Note that 1024 is an arbitrary number. This code probably needs
128  // improving to take the number of available cores into account.)
129  ptrdiff_t TaskSize = std::distance(Begin, End) / 1024;
130  if (TaskSize == 0)
131  TaskSize = 1;
132 
133  TaskGroup TG;
134  while (TaskSize < std::distance(Begin, End)) {
135  TG.spawn([=, &Fn] { std::for_each(Begin, Begin + TaskSize, Fn); });
136  Begin += TaskSize;
137  }
138  std::for_each(Begin, End, Fn);
139 }

141 template <class IndexTy, class FuncTy>
142 void parallel_for_each_n(IndexTy Begin, IndexTy End, FuncTy Fn) {
143  ptrdiff_t TaskSize = (End - Begin) / 1024;
144  if (TaskSize == 0)
145  TaskSize = 1;
146 
147  TaskGroup TG;
148  IndexTy I = Begin;
149  for (; I + TaskSize < End; I += TaskSize) {
150  TG.spawn([=, &Fn] {
151  for (IndexTy J = I, E = I + TaskSize; J != E; ++J)
152  Fn(J);
153  });
154  }
155  for (IndexTy J = I; J < End; ++J)
156  Fn(J);
157 }
