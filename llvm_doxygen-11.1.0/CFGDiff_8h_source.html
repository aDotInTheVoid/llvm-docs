26 // Two booleans are used to define orders in graphs:
27 // InverseGraph defines when we need to reverse the whole graph and is as such
28 // also equivalent to applying updates in reverse.
29 // InverseEdge defines whether we want to change the edges direction. E.g., for
30 // a non-inversed graph, the children are naturally the successors when
31 // InverseEdge is false and the predecessors when InverseEdge is true.
32 
33 // We define two base clases that call into GraphDiff, one for successors
34 // (CFGSuccessors), where InverseEdge is false, and one for predecessors
35 // (CFGPredecessors), where InverseEdge is true.
36 // FIXME: Further refactoring may merge the two base classes into a single one
37 // templated / parametrized on using succ_iterator/pred_iterator and false/true
38 // for the InverseEdge.
39 
40 // CFGViewChildren and CFGViewPredecessors, both can be parametrized to
41 // consider the graph inverted or not (i.e. InverseGraph). Successors
42 // implicitly has InverseEdge = false and Predecessors implicitly has
43 // InverseEdge = true (see calls to GraphDiff methods in there). The GraphTraits
44 // instantiations that follow define the value of InverseGraph.
45 
46 // GraphTraits instantiations:
47 // - GraphDiff<BasicBlock *> is equivalent to InverseGraph = false
48 // - GraphDiff<Inverse<BasicBlock *>> is equivalent to InverseGraph = true
49 // - second pair item is BasicBlock *, then InverseEdge = false (so it inherits
50 // from CFGViewChildren).
51 // - second pair item is Inverse<BasicBlock *>, then InverseEdge = true (so it
52 // inherits from CFGViewPredecessors).
53 
54 // The 4 GraphTraits are as follows:
55 // 1. std::pair<const GraphDiff<BasicBlock *> *, BasicBlock *>> :
56 // CFGViewChildren<false>
57 // Regular CFG, children means successors, InverseGraph = false,
58 // InverseEdge = false.
59 // 2. std::pair<const GraphDiff<Inverse<BasicBlock *>> *, BasicBlock *>> :
60 // CFGViewChildren<true>
61 // Reverse the graph, get successors but reverse-apply updates,
62 // InverseGraph = true, InverseEdge = false.
63 // 3. std::pair<const GraphDiff<BasicBlock *> *, Inverse<BasicBlock *>>> :
64 // CFGViewPredecessors<false>
65 // Regular CFG, reverse edges, so children mean predecessors,
66 // InverseGraph = false, InverseEdge = true.
67 // 4. std::pair<const GraphDiff<Inverse<BasicBlock *>> *, Inverse<BasicBlock *>>
68 // : CFGViewPredecessors<true>
69 // Reverse the graph and the edges, InverseGraph = true, InverseEdge = true.

73 // GraphDiff defines a CFG snapshot: given a set of Update<NodePtr>, provide
74 // utilities to skip edges marked as deleted and return a set of edges marked as
75 // newly inserted. The current diff treats the CFG as a graph rather than a
76 // multigraph. Added edges are pruned to be unique, and deleted edges will
77 // remove all existing edges between two blocks.
78 template <typename NodePtr, bool InverseGraph = false> class GraphDiff {
79  using UpdateMapType = SmallDenseMap<NodePtr, SmallVector<NodePtr, 2>>;
80  struct EdgesInsertedDeleted {
81  UpdateMapType Succ;
82  UpdateMapType Pred;
83  };
84  // Store Deleted edges on position 0, and Inserted edges on position 1.
85  EdgesInsertedDeleted Edges[2];
86  // By default, it is assumed that, given a CFG and a set of updates, we wish
87  // to apply these updates as given. If UpdatedAreReverseApplied is set, the
88  // updates will be applied in reverse: deleted edges are considered re-added
89  // and inserted edges are considered deleted when returning children.
90  bool UpdatedAreReverseApplied;
91  // Using a singleton empty vector for all node requests with no
92  // children.
93  SmallVector<NodePtr, 0> Empty;
94 
95  // Keep the list of legalized updates for a deterministic order of updates
96  // when using a GraphDiff for incremental updates in the DominatorTree.
97  // The list is kept in reverse to allow popping from end.
98  SmallVector<cfg::Update<NodePtr>, 4> LegalizedUpdates;
99 
100  void printMap(raw_ostream &OS, const UpdateMapType &M) const {
101  for (auto Pair : M)
102  for (auto Child : Pair.second) {
103  OS << "(";
104  Pair.first->printAsOperand(OS, false);
105  OS << ", ";
106  Child->printAsOperand(OS, false);
107  OS << ") ";
108  }
109  OS << "\n";
110  }
111 
112 public:
113  GraphDiff() : UpdatedAreReverseApplied(false) {}
114  GraphDiff(ArrayRef<cfg::Update<NodePtr>> Updates,
115  bool ReverseApplyUpdates = false) {
116  cfg::LegalizeUpdates<NodePtr>(Updates, LegalizedUpdates, InverseGraph,
117  /*ReverseResultOrder=*/true);
118  // The legalized updates are stored in reverse so we can pop_back when doing
119  // incremental updates.
120  for (auto U : LegalizedUpdates) {
121  unsigned IsInsert =
122  (U.getKind() == cfg::UpdateKind::Insert) == !ReverseApplyUpdates;
123  Edges[IsInsert].Succ[U.getFrom()].push_back(U.getTo());
124  Edges[IsInsert].Pred[U.getTo()].push_back(U.getFrom());
125  }
126  UpdatedAreReverseApplied = ReverseApplyUpdates;
127  }
128 
129  auto getLegalizedUpdates() const {
130  return make_range(LegalizedUpdates.begin(), LegalizedUpdates.end());
131  }
132 
133  unsigned getNumLegalizedUpdates() const { return LegalizedUpdates.size(); }
134 
135  cfg::Update<NodePtr> popUpdateForIncrementalUpdates() {
136  assert(!LegalizedUpdates.empty() && "No updates to apply!");
137  auto U = LegalizedUpdates.pop_back_val();
138  unsigned IsInsert =
139  (U.getKind() == cfg::UpdateKind::Insert) == !UpdatedAreReverseApplied;
140  auto &SuccList = Edges[IsInsert].Succ[U.getFrom()];
141  assert(SuccList.back() == U.getTo());
142  SuccList.pop_back();
143  if (SuccList.empty())
144  Edges[IsInsert].Succ.erase(U.getFrom());
145 
146  auto &PredList = Edges[IsInsert].Pred[U.getTo()];
147  assert(PredList.back() == U.getFrom());
148  PredList.pop_back();
149  if (PredList.empty())
150  Edges[IsInsert].Pred.erase(U.getTo());
151  return U;
152  }
153 
154  bool ignoreChild(const NodePtr BB, NodePtr EdgeEnd, bool InverseEdge) const {
155  // Used to filter nullptr in clang.
156  if (EdgeEnd == nullptr)
157  return true;
158  auto &DeleteChildren =
159  (InverseEdge != InverseGraph) ? Edges[0].Pred : Edges[0].Succ;
160  auto It = DeleteChildren.find(BB);
161  if (It == DeleteChildren.end())
162  return false;
163  auto &EdgesForBB = It->second;
164  return llvm::find(EdgesForBB, EdgeEnd) != EdgesForBB.end();
165  }
166 
167  iterator_range<typename SmallVectorImpl<NodePtr>::const_iterator>
168  getAddedChildren(const NodePtr BB, bool InverseEdge) const {
169  auto &InsertChildren =
170  (InverseEdge != InverseGraph) ? Edges[1].Pred : Edges[1].Succ;
171  auto It = InsertChildren.find(BB);
172  if (It == InsertChildren.end())
173  return make_range(Empty.begin(), Empty.end());
174  return make_range(It->second.begin(), It->second.end());
175  }
176 
177  void print(raw_ostream &OS) const {
178  OS << "===== GraphDiff: CFG edge changes to create a CFG snapshot. \n"
179  "===== (Note: notion of children/inverse_children depends on "
180  "the direction of edges and the graph.)\n";
181  OS << "Children to insert:\n\t";
182  printMap(OS, Edges[1].Succ);
183  OS << "Children to delete:\n\t";
184  printMap(OS, Edges[0].Succ);
185  OS << "Inverse_children to insert:\n\t";
186  printMap(OS, Edges[1].Pred);
187  OS << "Inverse_children to delete:\n\t";
188  printMap(OS, Edges[0].Pred);
189  OS << "\n";
190  }

194 #endif

197 template <typename GraphT, bool InverseGraph = false, bool InverseEdge = false,
198  typename GT = GraphTraits<GraphT>>
199 struct CFGViewChildren {
200  using DataRef = const GraphDiff<typename GT::NodeRef, InverseGraph> *;
201  using NodeRef = std::pair<DataRef, typename GT::NodeRef>;
202 
203  template<typename Range>
204  static auto makeChildRange(Range &&R, DataRef DR) {
205  using Iter = WrappedPairNodeDataIterator<decltype(std::forward<Range>(R).begin()), NodeRef, DataRef>;
206  return make_range(Iter(R.begin(), DR), Iter(R.end(), DR));
207  }
208 
209  static auto children(NodeRef N) {
210 
211  // filter iterator init:
212  auto R = make_range(GT::child_begin(N.second), GT::child_end(N.second));
213  // This lambda is copied into the iterators and persists to callers, ensure
214  // captures are by value or otherwise have sufficient lifetime.
215  auto First = make_filter_range(makeChildRange(R, N.first), [N](NodeRef C) {
216  return !C.first->ignoreChild(N.second, C.second, InverseEdge);
217  });
218 
219  // new inserts iterator init:
220  auto InsertVec = N.first->getAddedChildren(N.second, InverseEdge);
221  auto Second = makeChildRange(InsertVec, N.first);
222 
223  auto CR = concat<NodeRef>(First, Second);
224 
225  // concat_range contains references to other ranges, returning it would
226  // leave those references dangling - the iterators contain
227  // other iterators by value so they're safe to return.
228  return make_range(CR.begin(), CR.end());
229  }
230 
231  static auto child_begin(NodeRef N) {
232  return children(N).begin();
233  }
234 
235  static auto child_end(NodeRef N) {
236  return children(N).end();
237  }
238 
239  using ChildIteratorType = decltype(child_end(std::declval<NodeRef>()));
240 };
