70 /// GCOVBuffer - A wrapper around MemoryBuffer to provide GCOV specific
71 /// read operations.
72 class GCOVBuffer {
73 public:
74  GCOVBuffer(MemoryBuffer *B) : Buffer(B) {}
75  ~GCOVBuffer() { consumeError(cursor.takeError()); }
76 
77  /// readGCNOFormat - Check GCNO signature is valid at the beginning of buffer.
78  bool readGCNOFormat() {
79  StringRef buf = Buffer->getBuffer();
80  StringRef magic = buf.substr(0, 4);
81  if (magic == "gcno") {
82  de = DataExtractor(buf.substr(4), false, 0);
83  } else if (magic == "oncg") {
84  de = DataExtractor(buf.substr(4), true, 0);
85  } else {
86  errs() << "unexpected magic: " << magic << "\n";
87  return false;
88  }
89  return true;
90  }
91 
92  /// readGCDAFormat - Check GCDA signature is valid at the beginning of buffer.
93  bool readGCDAFormat() {
94  StringRef buf = Buffer->getBuffer();
95  StringRef magic = buf.substr(0, 4);
96  if (magic == "gcda") {
97  de = DataExtractor(buf.substr(4), false, 0);
98  } else if (magic == "adcg") {
99  de = DataExtractor(buf.substr(4), true, 0);
100  } else {
101  return false;
102  }
103  return true;
104  }
105 
106  /// readGCOVVersion - Read GCOV version.
107  bool readGCOVVersion(GCOV::GCOVVersion &Version) {
108  std::string str(de.getBytes(cursor, 4));
109  if (str.size() != 4)
110  return false;
111  if (de.isLittleEndian())
112  std::reverse(str.begin(), str.end());
113  int ver = str[0] >= 'A'
114  ? (str[0] - 'A') * 100 + (str[1] - '0') * 10 + str[2] - '0'
115  : (str[0] - '0') * 10 + str[2] - '0';
116  if (ver >= 90) {
117  // PR gcov-profile/84846, r269678
118  Version = GCOV::V900;
119  return true;
120  } else if (ver >= 80) {
121  // PR gcov-profile/48463
122  Version = GCOV::V800;
123  return true;
124  } else if (ver >= 48) {
125  // r189778: the exit block moved from the last to the second.
126  Version = GCOV::V408;
127  return true;
128  } else if (ver >= 47) {
129  // r173147: split checksum into cfg checksum and line checksum.
130  Version = GCOV::V407;
131  return true;
132  } else if (ver >= 34) {
133  Version = GCOV::V304;
134  return true;
135  }
136  errs() << "unexpected version: " << str << "\n";
137  return false;
138  }
139 
140  uint32_t getWord() { return de.getU32(cursor); }
141  StringRef getString() {
142  uint32_t len;
143  if (!readInt(len) || len == 0)
144  return {};
145  return de.getBytes(cursor, len * 4).split('\0').first;
146  }
147 
148  bool readInt(uint32_t &Val) {
149  if (cursor.tell() + 4 > de.size()) {
150  Val = 0;
151  errs() << "unexpected end of memory buffer: " << cursor.tell() << "\n";
152  return false;
153  }
154  Val = de.getU32(cursor);
155  return true;
156  }
157 
158  bool readInt64(uint64_t &Val) {
159  uint32_t Lo, Hi;
160  if (!readInt(Lo) || !readInt(Hi))
161  return false;
162  Val = ((uint64_t)Hi << 32) | Lo;
163  return true;
164  }
165 
166  bool readString(StringRef &Str) {
167  uint32_t len;
168  if (!readInt(len) || len == 0)
169  return false;
170  Str = de.getBytes(cursor, len * 4).split('\0').first;
171  return bool(cursor);
172  }
173 
174  DataExtractor de{ArrayRef<uint8_t>{}, false, 0};
175  DataExtractor::Cursor cursor{0};
176 
177 private:
178  MemoryBuffer *Buffer;
179 };
