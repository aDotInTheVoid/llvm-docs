22 template <typename ValueType, typename Enumeration,

28  EnumeratedArray(ValueType V) {
29  for (IndexType IX = 0; IX < Size; ++IX) {
30  Underlying[IX] = V;
31  }
32  }
33  inline const ValueType &operator[](const Enumeration Index) const {

37  }
