71 // Register debug object, return error message or null for success.
72 static void registerJITLoaderGDBImpl(JITTargetAddress Addr, uint64_t Size) {
73  jit_code_entry *E = new jit_code_entry;
74  E->symfile_addr = jitTargetAddressToPointer<const char *>(Addr);
75  E->symfile_size = Size;
76  E->prev_entry = nullptr;
77 
78  std::lock_guard<std::mutex> Lock(*JITDebugLock);
79 
80  // Insert this entry at the head of the list.
81  jit_code_entry *NextEntry = __jit_debug_descriptor.first_entry;
82  E->next_entry = NextEntry;
83  if (NextEntry) {
84  NextEntry->prev_entry = E;
85  }
86 
87  __jit_debug_descriptor.first_entry = E;
88  __jit_debug_descriptor.relevant_entry = E;
89 
90  // Run into the rendezvous breakpoint.
91  __jit_debug_descriptor.action_flag = JIT_REGISTER_FN;
92  __jit_debug_register_code();
93 }
