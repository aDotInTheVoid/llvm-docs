76 ConstantRange addOverflowNever(const ConstantRange &L, const ConstantRange &R) {
77  assert(!L.isSignWrappedSet());
78  assert(!R.isSignWrappedSet());
79  if (L.signedAddMayOverflow(R) !=
80  ConstantRange::OverflowResult::NeverOverflows)
81  return ConstantRange::getFull(L.getBitWidth());
82  ConstantRange Result = L.add(R);
83  assert(!Result.isSignWrappedSet());
84  return Result;
85 }
86 
87 ConstantRange unionNoWrap(const ConstantRange &L, const ConstantRange &R) {
88  assert(!L.isSignWrappedSet());
89  assert(!R.isSignWrappedSet());
90  auto Result = L.unionWith(R);
91  // Two non-wrapped sets can produce wrapped.
92  if (Result.isSignWrappedSet())
93  Result = ConstantRange::getFull(Result.getBitWidth());
94  return Result;
95 }

144 /// Calculate the allocation size of a given alloca. Returns empty range
145 // in case of confution.
146 ConstantRange getStaticAllocaSizeRange(const AllocaInst &AI) {
147  const DataLayout &DL = AI.getModule()->getDataLayout();
148  TypeSize TS = DL.getTypeAllocSize(AI.getAllocatedType());
149  unsigned PointerSize = DL.getMaxPointerSizeInBits();
150  // Fallback to empty range for alloca size.
151  ConstantRange R = ConstantRange::getEmpty(PointerSize);
152  if (TS.isScalable())
153  return R;
154  APInt APSize(PointerSize, TS.getFixedSize(), true);
155  if (APSize.isNonPositive())
156  return R;
157  if (AI.isArrayAllocation()) {
158  const auto *C = dyn_cast<ConstantInt>(AI.getArraySize());
159  if (!C)
160  return R;
161  bool Overflow = false;
162  APInt Mul = C->getValue();
163  if (Mul.isNonPositive())
164  return R;
165  Mul = Mul.sextOrTrunc(PointerSize);
166  APSize = APSize.smul_ov(Mul, Overflow);
167  if (Overflow)
168  return R;
169  }
170  R = ConstantRange(APInt::getNullValue(PointerSize), APSize);
171  assert(!isUnsafe(R));
172  return R;
173 }

258 ConstantRange StackSafetyLocalAnalysis::offsetFrom(Value *Addr, Value *Base) {
259  if (!SE.isSCEVable(Addr->getType()) || !SE.isSCEVable(Base->getType()))
260  return UnknownRange;
261 
262  auto *PtrTy = IntegerType::getInt8PtrTy(SE.getContext());
263  const SCEV *AddrExp = SE.getTruncateOrZeroExtend(SE.getSCEV(Addr), PtrTy);
264  const SCEV *BaseExp = SE.getTruncateOrZeroExtend(SE.getSCEV(Base), PtrTy);
265  const SCEV *Diff = SE.getMinusSCEV(AddrExp, BaseExp);
266  if (isa<SCEVCouldNotCompute>(Diff))
267  return UnknownRange;
268 
269  ConstantRange Offset = SE.getSignedRange(Diff);
270  if (isUnsafe(Offset))
271  return UnknownRange;
272  return Offset.sextOrTrunc(PointerSize);
273 }

275 ConstantRange
276 StackSafetyLocalAnalysis::getAccessRange(Value *Addr, Value *Base,
277  const ConstantRange &SizeRange) {
278  // Zero-size loads and stores do not access memory.
279  if (SizeRange.isEmptySet())
280  return ConstantRange::getEmpty(PointerSize);
281  assert(!isUnsafe(SizeRange));
282 
283  ConstantRange Offsets = offsetFrom(Addr, Base);
284  if (isUnsafe(Offsets))
285  return UnknownRange;
286 
287  Offsets = addOverflowNever(Offsets, SizeRange);
288  if (isUnsafe(Offsets))
289  return UnknownRange;
290  return Offsets;

304 ConstantRange StackSafetyLocalAnalysis::getMemIntrinsicAccessRange(
305  const MemIntrinsic *MI, const Use &U, Value *Base) {
306  if (const auto *MTI = dyn_cast<MemTransferInst>(MI)) {
307  if (MTI->getRawSource() != U && MTI->getRawDest() != U)
308  return ConstantRange::getEmpty(PointerSize);
309  } else {
310  if (MI->getRawDest() != U)
311  return ConstantRange::getEmpty(PointerSize);
312  }
313 
314  auto *CalculationTy = IntegerType::getIntNTy(SE.getContext(), PointerSize);
315  if (!SE.isSCEVable(MI->getLength()->getType()))
316  return UnknownRange;
317 
318  const SCEV *Expr =
319  SE.getTruncateOrZeroExtend(SE.getSCEV(MI->getLength()), CalculationTy);
320  ConstantRange Sizes = SE.getSignedRange(Expr);
321  if (Sizes.getUpper().isNegative() || isUnsafe(Sizes))
322  return UnknownRange;
323  Sizes = Sizes.sextOrTrunc(PointerSize);
324  ConstantRange SizeRange(APInt::getNullValue(PointerSize),
325  Sizes.getUpper() - 1);
326  return getAccessRange(U, Base, SizeRange);

329 /// The function analyzes all local uses of Ptr (alloca or argument) and
330 /// calculates local access range and all function calls where it was used.
331 bool StackSafetyLocalAnalysis::analyzeAllUses(Value *Ptr,
332  UseInfo<GlobalValue> &US,
333  const StackLifetime &SL) {
334  SmallPtrSet<const Value *, 16> Visited;
335  SmallVector<const Value *, 8> WorkList;
336  WorkList.push_back(Ptr);
337  const AllocaInst *AI = dyn_cast<AllocaInst>(Ptr);
338 
339  // A DFS search through all uses of the alloca in bitcasts/PHI/GEPs/etc.
340  while (!WorkList.empty()) {
341  const Value *V = WorkList.pop_back_val();
342  for (const Use &UI : V->uses()) {
343  const auto *I = cast<Instruction>(UI.getUser());
344  if (!SL.isReachable(I))
345  continue;
346 
347  assert(V == UI.get());
348 
349  switch (I->getOpcode()) {
350  case Instruction::Load: {
351  if (AI && !SL.isAliveAfter(AI, I)) {
352  US.updateRange(UnknownRange);
353  return false;
354  }
355  US.updateRange(
356  getAccessRange(UI, Ptr, DL.getTypeStoreSize(I->getType())));
357  break;
358  }
359 
360  case Instruction::VAArg:
361  // "va-arg" from a pointer is safe.
362  break;
363  case Instruction::Store: {
364  if (V == I->getOperand(0)) {
365  // Stored the pointer - conservatively assume it may be unsafe.
366  US.updateRange(UnknownRange);
367  return false;
368  }
369  if (AI && !SL.isAliveAfter(AI, I)) {
370  US.updateRange(UnknownRange);
371  return false;
372  }
373  US.updateRange(getAccessRange(
374  UI, Ptr, DL.getTypeStoreSize(I->getOperand(0)->getType())));
375  break;
376  }
377 
378  case Instruction::Ret:
379  // Information leak.
380  // FIXME: Process parameters correctly. This is a leak only if we return
381  // alloca.
382  US.updateRange(UnknownRange);
383  return false;
384 
385  case Instruction::Call:
386  case Instruction::Invoke: {
387  if (I->isLifetimeStartOrEnd())
388  break;
389 
390  if (AI && !SL.isAliveAfter(AI, I)) {
391  US.updateRange(UnknownRange);
392  return false;
393  }
394 
395  if (const MemIntrinsic *MI = dyn_cast<MemIntrinsic>(I)) {
396  US.updateRange(getMemIntrinsicAccessRange(MI, UI, Ptr));
397  break;
398  }
399 
400  const auto &CB = cast<CallBase>(*I);
401  if (!CB.isArgOperand(&UI)) {
402  US.updateRange(UnknownRange);
403  return false;
404  }
405 
406  unsigned ArgNo = CB.getArgOperandNo(&UI);
407  if (CB.isByValArgument(ArgNo)) {
408  US.updateRange(getAccessRange(
409  UI, Ptr, DL.getTypeStoreSize(CB.getParamByValType(ArgNo))));
410  break;
411  }
412 
413  // FIXME: consult devirt?
414  // Do not follow aliases, otherwise we could inadvertently follow
415  // dso_preemptable aliases or aliases with interposable linkage.
416  const GlobalValue *Callee =
417  dyn_cast<GlobalValue>(CB.getCalledOperand()->stripPointerCasts());
418  if (!Callee) {
419  US.updateRange(UnknownRange);
420  return false;
421  }
422 
423  assert(isa<Function>(Callee) || isa<GlobalAlias>(Callee));
424  ConstantRange Offsets = offsetFrom(UI, Ptr);
425  auto Insert =
426  US.Calls.emplace(CallInfo<GlobalValue>(Callee, ArgNo), Offsets);
427  if (!Insert.second)
428  Insert.first->second = Insert.first->second.unionWith(Offsets);
429  break;
430  }
431 
432  default:
433  if (Visited.insert(I).second)
434  WorkList.push_back(cast<const Instruction>(I));
435  }
436  }
437  }
438 
439  return true;
440 }

442 FunctionInfo<GlobalValue> StackSafetyLocalAnalysis::run() {
443  FunctionInfo<GlobalValue> Info;
444  assert(!F.isDeclaration() &&
445  "Can't run StackSafety on a function declaration");
446 
447  LLVM_DEBUG(dbgs() << "[StackSafety] " << F.getName() << "\n");
448 
449  SmallVector<AllocaInst *, 64> Allocas;
450  for (auto &I : instructions(F))
451  if (auto *AI = dyn_cast<AllocaInst>(&I))
452  Allocas.push_back(AI);
453  StackLifetime SL(F, Allocas, StackLifetime::LivenessType::Must);
454  SL.run();
455 
456  for (auto *AI : Allocas) {
457  auto &UI = Info.Allocas.emplace(AI, PointerSize).first->second;
458  analyzeAllUses(AI, UI, SL);
459  }
460 
461  for (Argument &A : F.args()) {
462  // Non pointers and bypass arguments are not going to be used in any global
463  // processing.
464  if (A.getType()->isPointerTy() && !A.hasByValAttr()) {
465  auto &UI = Info.Params.emplace(A.getArgNo(), PointerSize).first->second;
466  analyzeAllUses(&A, UI, SL);
467  }
468  }
469 
470  LLVM_DEBUG(Info.print(dbgs(), F.getName(), &F));
471  LLVM_DEBUG(dbgs() << "[StackSafety] done\n");
472  return Info;
473 }

510 template <typename CalleeTy>
511 ConstantRange StackSafetyDataFlowAnalysis<CalleeTy>::getArgumentAccessRange(
512  const CalleeTy *Callee, unsigned ParamNo,
513  const ConstantRange &Offsets) const {
514  auto FnIt = Functions.find(Callee);
515  // Unknown callee (outside of LTO domain or an indirect call).
516  if (FnIt == Functions.end())
517  return UnknownRange;
518  auto &FS = FnIt->second;
519  auto ParamIt = FS.Params.find(ParamNo);
520  if (ParamIt == FS.Params.end())
521  return UnknownRange;
522  auto &Access = ParamIt->second.Range;
523  if (Access.isEmptySet())
524  return Access;
525  if (Access.isFullSet())
526  return UnknownRange;
527  return addOverflowNever(Access, Offsets);
528 }

530 template <typename CalleeTy>
531 bool StackSafetyDataFlowAnalysis<CalleeTy>::updateOneUse(UseInfo<CalleeTy> &US,
532  bool UpdateToFullSet) {
533  bool Changed = false;
534  for (auto &KV : US.Calls) {
535  assert(!KV.second.isEmptySet() &&
536  "Param range can't be empty-set, invalid offset range");
537 
538  ConstantRange CalleeRange =
539  getArgumentAccessRange(KV.first.Callee, KV.first.ParamNo, KV.second);
540  if (!US.Range.contains(CalleeRange)) {
541  Changed = true;
542  if (UpdateToFullSet)
543  US.Range = UnknownRange;
544  else
545  US.updateRange(CalleeRange);
546  }
547  }
548  return Changed;
549 }

551 template <typename CalleeTy>
552 void StackSafetyDataFlowAnalysis<CalleeTy>::updateOneNode(
553  const CalleeTy *Callee, FunctionInfo<CalleeTy> &FS) {
554  bool UpdateToFullSet = FS.UpdateCount > StackSafetyMaxIterations;
555  bool Changed = false;
556  for (auto &KV : FS.Params)
557  Changed |= updateOneUse(KV.second, UpdateToFullSet);
558 
559  if (Changed) {
560  LLVM_DEBUG(dbgs() << "=== update [" << FS.UpdateCount
561  << (UpdateToFullSet ? ", full-set" : "") << "] " << &FS
562  << "\n");
563  // Callers of this function may need updating.
564  for (auto &CallerID : Callers[Callee])
565  WorkList.insert(CallerID);
566 
567  ++FS.UpdateCount;
568  }

571 template <typename CalleeTy>
572 void StackSafetyDataFlowAnalysis<CalleeTy>::runDataFlow() {
573  SmallVector<const CalleeTy *, 16> Callees;
574  for (auto &F : Functions) {
575  Callees.clear();
576  auto &FS = F.second;
577  for (auto &KV : FS.Params)
578  for (auto &CS : KV.second.Calls)
579  Callees.push_back(CS.first.Callee);
580 
581  llvm::sort(Callees);
582  Callees.erase(std::unique(Callees.begin(), Callees.end()), Callees.end());
583 
584  for (auto &Callee : Callees)
585  Callers[Callee].push_back(F.first);
586  }
587 
588  updateAllNodes();
589 
590  while (!WorkList.empty()) {
591  const CalleeTy *Callee = WorkList.back();
592  WorkList.pop_back();
593  updateOneNode(Callee);
594  }
595 }

614 FunctionSummary *findCalleeFunctionSummary(ValueInfo VI, StringRef ModuleId) {
615  if (!VI)
616  return nullptr;
617  auto SummaryList = VI.getSummaryList();
618  GlobalValueSummary* S = nullptr;
619  for (const auto& GVS : SummaryList) {
620  if (!GVS->isLive())
621  continue;
622  if (const AliasSummary *AS = dyn_cast<AliasSummary>(GVS.get()))
623  if (!AS->hasAliasee())
624  continue;
625  if (!isa<FunctionSummary>(GVS->getBaseObject()))
626  continue;
627  if (GlobalValue::isLocalLinkage(GVS->linkage())) {
628  if (GVS->modulePath() == ModuleId) {
629  S = GVS.get();
630  break;
631  }
632  } else if (GlobalValue::isExternalLinkage(GVS->linkage())) {
633  if (S) {
634  ++NumIndexCalleeMultipleExternal;
635  return nullptr;
636  }
637  S = GVS.get();
638  } else if (GlobalValue::isWeakLinkage(GVS->linkage())) {
639  if (S) {
640  ++NumIndexCalleeMultipleWeak;
641  return nullptr;
642  }
643  S = GVS.get();
644  } else if (GlobalValue::isAvailableExternallyLinkage(GVS->linkage()) ||
645  GlobalValue::isLinkOnceLinkage(GVS->linkage())) {
646  if (SummaryList.size() == 1)
647  S = GVS.get();
