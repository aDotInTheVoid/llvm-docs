103  return (TSFlags & InstFormatMask) >> InstFormatShift;

206  return "rdn";
207  case RISCVFPRndMode::RUP:
208  return "rup";

216 inline static RoundingMode stringToRoundingMode(StringRef Str) {

220  .Case("rdn", RISCVFPRndMode::RDN)
