40 void yaml::BinaryRef::writeAsBinary(raw_ostream &OS, uint64_t N) const {
41  if (!DataIsHexString) {
42  OS.write((const char *)Data.data(), std::min<uint64_t>(N, Data.size()));
43  return;
44  }
45 
46  for (uint64_t I = 0, E = std::min<uint64_t>(N, Data.size() / 2); I != E;
47  ++I) {
48  uint8_t Byte = llvm::hexDigitValue(Data[I * 2]);
49  Byte <<= 4;
50  Byte |= llvm::hexDigitValue(Data[I * 2 + 1]);
51  OS.write(Byte);
52  }
53 }

55 void yaml::BinaryRef::writeAsHex(raw_ostream &OS) const {
