116 void WebAssemblyTargetWasmStreamer::emitLocal(ArrayRef<wasm::ValType> Types) {
117  SmallVector<std::pair<wasm::ValType, uint32_t>, 4> Grouped;
118  for (auto Type : Types) {
119  if (Grouped.empty() || Grouped.back().first != Type)
120  Grouped.push_back(std::make_pair(Type, 1));
121  else
122  ++Grouped.back().second;
123  }
124 
125  Streamer.emitULEB128IntValue(Grouped.size());
126  for (auto Pair : Grouped) {
127  Streamer.emitULEB128IntValue(Pair.second);
128  emitValueType(Pair.first);
129  }
130 }
