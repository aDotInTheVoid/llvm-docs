94 MCOperand MSP430MCInstLower::
95 LowerSymbolOperand(const MachineOperand &MO, MCSymbol *Sym) const {
96  // FIXME: We would like an efficient form for this, so we don't have to do a
97  // lot of extra uniquing.
98  const MCExpr *Expr = MCSymbolRefExpr::create(Sym, Ctx);
99 
100  switch (MO.getTargetFlags()) {
101  default: llvm_unreachable("Unknown target flag on GV operand");
102  case 0: break;
103  }
104 
105  if (!MO.isJTI() && MO.getOffset())
106  Expr = MCBinaryExpr::createAdd(Expr,
107  MCConstantExpr::create(MO.getOffset(), Ctx),
108  Ctx);
109  return MCOperand::createExpr(Expr);
110 }

115 void MSP430MCInstLower::Lower(const MachineInstr *MI, MCInst &OutMI) const {
116  OutMI.setOpcode(MI->getOpcode());
117 
118  for (unsigned i = 0, e = MI->getNumOperands(); i != e; ++i) {
119  const MachineOperand &MO = MI->getOperand(i);
120 
121  MCOperand MCOp;
122  switch (MO.getType()) {
123  default:
124  MI->print(errs());
125  llvm_unreachable("unknown operand type");
126  case MachineOperand::MO_Register:
127  // Ignore all implicit register operands.
128  if (MO.isImplicit()) continue;
129  MCOp = MCOperand::createReg(MO.getReg());
130  break;
131  case MachineOperand::MO_Immediate:
132  MCOp = MCOperand::createImm(MO.getImm());
133  break;
134  case MachineOperand::MO_MachineBasicBlock:
135  MCOp = MCOperand::createExpr(MCSymbolRefExpr::create(
136  MO.getMBB()->getSymbol(), Ctx));
137  break;
138  case MachineOperand::MO_GlobalAddress:
139  MCOp = LowerSymbolOperand(MO, GetGlobalAddressSymbol(MO));
140  break;
141  case MachineOperand::MO_ExternalSymbol:
142  MCOp = LowerSymbolOperand(MO, GetExternalSymbolSymbol(MO));
143  break;
144  case MachineOperand::MO_JumpTableIndex:
145  MCOp = LowerSymbolOperand(MO, GetJumpTableSymbol(MO));
146  break;
147  case MachineOperand::MO_ConstantPoolIndex:
148  MCOp = LowerSymbolOperand(MO, GetConstantPoolIndexSymbol(MO));
149  break;
150  case MachineOperand::MO_BlockAddress:
151  MCOp = LowerSymbolOperand(MO, GetBlockAddressSymbol(MO));
152  break;
153  case MachineOperand::MO_RegisterMask:
154  continue;
155  }
156 
157  OutMI.addOperand(MCOp);
158  }
159 }
