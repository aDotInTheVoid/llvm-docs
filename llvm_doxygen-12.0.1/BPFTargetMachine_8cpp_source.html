104 void BPFTargetMachine::adjustPassManager(PassManagerBuilder &Builder) {
105  Builder.addExtension(
106  PassManagerBuilder::EP_EarlyAsPossible,
107  [&](const PassManagerBuilder &, legacy::PassManagerBase &PM) {
108  PM.add(createBPFAbstractMemberAccess(this));
109  PM.add(createBPFPreserveDIType());
110  });
111 
112  Builder.addExtension(
113  PassManagerBuilder::EP_Peephole,
114  [&](const PassManagerBuilder &, legacy::PassManagerBase &PM) {
115  PM.add(createCFGSimplificationPass(
116  SimplifyCFGOptions().hoistCommonInsts(true)));
117  });
118  Builder.addExtension(
119  PassManagerBuilder::EP_ModuleOptimizerEarly,
120  [&](const PassManagerBuilder &, legacy::PassManagerBase &PM) {
121  PM.add(createBPFAdjustOpt());
122  });
123 }

125 void BPFTargetMachine::registerPassBuilderCallbacks(PassBuilder &PB,
126  bool DebugPassManager) {
127  PB.registerPipelineStartEPCallback(
128  [=](ModulePassManager &MPM, PassBuilder::OptimizationLevel) {
129  FunctionPassManager FPM(DebugPassManager);
130  FPM.addPass(BPFAbstractMemberAccessPass(this));
131  FPM.addPass(BPFPreserveDITypePass());
132  MPM.addPass(createModuleToFunctionPassAdaptor(std::move(FPM)));
133  });
134  PB.registerPeepholeEPCallback([=](FunctionPassManager &FPM,
135  PassBuilder::OptimizationLevel Level) {
136  FPM.addPass(SimplifyCFGPass(SimplifyCFGOptions().hoistCommonInsts(true)));
137  });
138  PB.registerPipelineEarlySimplificationEPCallback(
139  [=](ModulePassManager &MPM, PassBuilder::OptimizationLevel) {
140  MPM.addPass(BPFAdjustOptPass());
141  });
142 }

162 void BPFPassConfig::addMachineSSAOptimization() {
163  addPass(createBPFMISimplifyPatchablePass());
164 
165  // The default implementation must be called first as we want eBPF
166  // Peephole ran at last.
167  TargetPassConfig::addMachineSSAOptimization();
168 
169  const BPFSubtarget *Subtarget = getBPFTargetMachine().getSubtargetImpl();
170  if (!DisableMIPeephole) {
171  if (Subtarget->getHasAlu32())
172  addPass(createBPFMIPeepholePass());
173  addPass(createBPFMIPeepholeTruncElimPass());
174  }
175 }
