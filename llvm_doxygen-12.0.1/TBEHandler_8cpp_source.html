72 /// YAML traits for ELFSymbol.

75  IO.mapRequired("Type", Symbol.Type);
76  // The need for symbol size depends on the symbol type.
77  if (Symbol.Type == ELFSymbolType::NoType) {
78  IO.mapOptional("Size", Symbol.Size, (uint64_t)0);
