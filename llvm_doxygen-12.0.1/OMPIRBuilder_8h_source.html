197  /// Generator for the control flow structure of an OpenMP canonical loop.
198  ///
199  /// This generator operates on the logical iteration space of the loop, i.e.
200  /// the caller only has to provide a loop trip count of the loop as defined by
201  /// base language semantics. The trip count is interpreted as an unsigned
202  /// integer. The induction variable passed to \p BodyGenCB will be of the same
203  /// type and run from 0 to \p TripCount - 1. It is up to the callback to
204  /// convert the logical iteration variable to the loop counter variable in the
205  /// loop body.
206  ///
207  /// \param Loc The insert and source location description. The insert
208  /// location can be between two instructions or the end of a
209  /// degenerate block (e.g. a BB under construction).
210  /// \param BodyGenCB Callback that will generate the loop body code.
211  /// \param TripCount Number of iterations the loop body is executed.
212  /// \param Name Base name used to derive BB and instruction names.
213  ///
214  /// \returns An object representing the created control flow structure which
215  /// can be used for loop-associated directives.
216  CanonicalLoopInfo *createCanonicalLoop(const LocationDescription &Loc,
217  LoopBodyGenCallbackTy BodyGenCB,
218  Value *TripCount,
219  const Twine &Name = "loop");

221  /// Generator for the control flow structure of an OpenMP canonical loop.
222  ///
223  /// Instead of a logical iteration space, this allows specifying user-defined
224  /// loop counter values using increment, upper- and lower bounds. To
225  /// disambiguate the terminology when counting downwards, instead of lower
226  /// bounds we use \p Start for the loop counter value in the first body
227  /// iteration.
228  ///
229  /// Consider the following limitations:
230  ///
231  /// * A loop counter space over all integer values of its bit-width cannot be
232  /// represented. E.g using uint8_t, its loop trip count of 256 cannot be
233  /// stored into an 8 bit integer):
234  ///
235  /// DO I = 0, 255, 1
236  ///
237  /// * Unsigned wrapping is only supported when wrapping only "once"; E.g.
238  /// effectively counting downwards:
239  ///
240  /// for (uint8_t i = 100u; i > 0; i += 127u)
241  ///
242  ///
243  /// TODO: May need to add additional parameters to represent:
244  ///
245  /// * Allow representing downcounting with unsigned integers.
246  ///
247  /// * Sign of the step and the comparison operator might disagree:
248  ///
249  /// for (int i = 0; i < 42; --i)
250  ///
251  //
252  /// \param Loc The insert and source location description.
253  /// \param BodyGenCB Callback that will generate the loop body code.
254  /// \param Start Value of the loop counter for the first iterations.
255  /// \param Stop Loop counter values past this will stop the the
256  /// iterations.
257  /// \param Step Loop counter increment after each iteration; negative
258  /// means counting down. \param IsSigned Whether Start, Stop
259  /// and Stop are signed integers.
260  /// \param InclusiveStop Whether \p Stop itself is a valid value for the loop
261  /// counter.
262  /// \param ComputeIP Insertion point for instructions computing the trip
263  /// count. Can be used to ensure the trip count is available
264  /// at the outermost loop of a loop nest. If not set,
265  /// defaults to the preheader of the generated loop.
266  /// \param Name Base name used to derive BB and instruction names.
267  ///
268  /// \returns An object representing the created control flow structure which
269  /// can be used for loop-associated directives.
270  CanonicalLoopInfo *createCanonicalLoop(const LocationDescription &Loc,
271  LoopBodyGenCallbackTy BodyGenCB,
272  Value *Start, Value *Stop, Value *Step,
273  bool IsSigned, bool InclusiveStop,
274  InsertPointTy ComputeIP = {},
275  const Twine &Name = "loop");

303  /// Tile a loop nest.
304  ///
305  /// Tiles the loops of \p Loops by the tile sizes in \p TileSizes. Loops in
306  /// \p/ Loops must be perfectly nested, from outermost to innermost loop
307  /// (i.e. Loops.front() is the outermost loop). The trip count llvm::Value
308  /// of every loop and every tile sizes must be usable in the outermost
309  /// loop's preheader. This implies that the loop nest is rectangular.
310  ///
311  /// Example:
312  /// \code
313  /// for (int i = 0; i < 15; ++i) // Canonical loop "i"
314  /// for (int j = 0; j < 14; ++j) // Canonical loop "j"
315  /// body(i, j);
316  /// \endcode
317  ///
318  /// After tiling with Loops={i,j} and TileSizes={5,7}, the loop is changed to
319  /// \code
320  /// for (int i1 = 0; i1 < 3; ++i1)
321  /// for (int j1 = 0; j1 < 2; ++j1)
322  /// for (int i2 = 0; i2 < 5; ++i2)
323  /// for (int j2 = 0; j2 < 7; ++j2)
324  /// body(i1*3+i2, j1*3+j2);
325  /// \endcode
326  ///
327  /// The returned vector are the loops {i1,j1,i2,j2}. The loops i1 and j1 are
328  /// referred to the floor, and the loops i2 and j2 are the tiles. Tiling also
329  /// handles non-constant trip counts, non-constant tile sizes and trip counts
330  /// that are not multiples of the tile size. In the latter case the tile loop
331  /// of the last floor-loop iteration will have fewer iterations than specified
332  /// as its tile size.
333  ///
334  ///
335  /// @param DL Debug location for instructions added by tiling, for
336  /// instance the floor- and tile trip count computation.
337  /// @param Loops Loops to tile. The CanonicalLoopInfo objects are
338  /// invalidated by this method, i.e. should not used after
339  /// tiling.
340  /// @param TileSizes For each loop in \p Loops, the tile size for that
341  /// dimensions.
342  ///
343  /// \returns A list of generated loops. Contains twice as many loops as the
344  /// input loop nest; the first half are the floor loops and the
345  /// second half are the tile loops.
346  std::vector<CanonicalLoopInfo *>
347  tileLoops(DebugLoc DL, ArrayRef<CanonicalLoopInfo *> Loops,
348  ArrayRef<Value *> TileSizes);

732 /// Class to represented the control flow structure of an OpenMP canonical loop.
733 ///
734 /// The control-flow structure is standardized for easy consumption by
735 /// directives associated with loops. For instance, the worksharing-loop
736 /// construct may change this control flow such that each loop iteration is
737 /// executed on only one thread.
738 ///
739 /// The control flow can be described as follows:
740 ///
741 /// Preheader
742 /// |
743 /// /-> Header
744 /// | |
745 /// | Cond---\
746 /// | | |
747 /// | Body |
748 /// | | | |
749 /// | <...> |
750 /// | | | |
751 /// \--Latch |
752 /// |
753 /// Exit
754 /// |
755 /// After
756 ///
757 /// Code in the header, condition block, latch and exit block must not have any
758 /// side-effect. The body block is the single entry point into the loop body,
759 /// which may contain arbitrary control flow as long as all control paths
760 /// eventually branch to the latch block.
761 ///
762 /// Defined outside OpenMPIRBuilder because one cannot forward-declare nested
763 /// classes.
764 class CanonicalLoopInfo {
765  friend class OpenMPIRBuilder;
766 
767 private:
768  /// Whether this object currently represents a loop.
769  bool IsValid = false;
770 
771  BasicBlock *Preheader;
772  BasicBlock *Header;
773  BasicBlock *Cond;
774  BasicBlock *Body;
775  BasicBlock *Latch;
776  BasicBlock *Exit;
777  BasicBlock *After;
778 
779  /// Add the control blocks of this loop to \p BBs.
780  ///
781  /// This does not include any block from the body, including the one returned
782  /// by getBody().
783  void collectControlBlocks(SmallVectorImpl<BasicBlock *> &BBs);
784 
785 public:
786  /// The preheader ensures that there is only a single edge entering the loop.
787  /// Code that must be execute before any loop iteration can be emitted here,
788  /// such as computing the loop trip count and begin lifetime markers. Code in
789  /// the preheader is not considered part of the canonical loop.
790  BasicBlock *getPreheader() const { return Preheader; }
791 
792  /// The header is the entry for each iteration. In the canonical control flow,
793  /// it only contains the PHINode for the induction variable.
794  BasicBlock *getHeader() const { return Header; }
795 
796  /// The condition block computes whether there is another loop iteration. If
797  /// yes, branches to the body; otherwise to the exit block.
798  BasicBlock *getCond() const { return Cond; }
799 
800  /// The body block is the single entry for a loop iteration and not controlled
801  /// by CanonicalLoopInfo. It can contain arbitrary control flow but must
802  /// eventually branch to the \p Latch block.
803  BasicBlock *getBody() const { return Body; }
804 
805  /// Reaching the latch indicates the end of the loop body code. In the
806  /// canonical control flow, it only contains the increment of the induction
807  /// variable.
808  BasicBlock *getLatch() const { return Latch; }
809 
810  /// Reaching the exit indicates no more iterations are being executed.
811  BasicBlock *getExit() const { return Exit; }
812 
813  /// The after block is intended for clean-up code such as lifetime end
814  /// markers. It is separate from the exit block to ensure, analogous to the
815  /// preheader, it having just a single entry edge and being free from PHI
816  /// nodes should there be multiple loop exits (such as from break
817  /// statements/cancellations).
818  BasicBlock *getAfter() const { return After; }
819 
820  /// Returns the llvm::Value containing the number of loop iterations. It must
821  /// be valid in the preheader and always interpreted as an unsigned integer of
822  /// any bit-width.
823  Value *getTripCount() const {
824  Instruction *CmpI = &Cond->front();
825  assert(isa<CmpInst>(CmpI) && "First inst must compare IV with TripCount");
826  return CmpI->getOperand(1);
827  }
828 
829  /// Returns the instruction representing the current logical induction
830  /// variable. Always unsigned, always starting at 0 with an increment of one.
831  Instruction *getIndVar() const {
832  Instruction *IndVarPHI = &Header->front();
833  assert(isa<PHINode>(IndVarPHI) && "First inst must be the IV PHI");
834  return IndVarPHI;
835  }
836 
837  /// Return the type of the induction variable (and the trip count).
838  Type *getIndVarType() const { return getIndVar()->getType(); }
839 
840  /// Return the insertion point for user code before the loop.
841  OpenMPIRBuilder::InsertPointTy getPreheaderIP() const {
842  return {Preheader, std::prev(Preheader->end())};
843  };
844 
845  /// Return the insertion point for user code in the body.
846  OpenMPIRBuilder::InsertPointTy getBodyIP() const {
847  return {Body, Body->begin()};
848  };
849 
850  /// Return the insertion point for user code after the loop.
851  OpenMPIRBuilder::InsertPointTy getAfterIP() const {
852  return {After, After->begin()};
853  };
854 
855  /// Consistency self-check.
856  void assertOK() const;
857 };
