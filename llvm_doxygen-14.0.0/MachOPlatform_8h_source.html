50 /// Mediates between MachO initialization and ExecutionSession state.
51 class MachOPlatform : public Platform {
52 public:
53  /// Try to create a MachOPlatform instance, adding the ORC runtime to the
54  /// given JITDylib.
55  ///
56  /// The ORC runtime requires access to a number of symbols in libc++, and
57  /// requires access to symbols in libobjc, and libswiftCore to support
58  /// Objective-C and Swift code. It is up to the caller to ensure that the
59  /// requried symbols can be referenced by code added to PlatformJD. The
60  /// standard way to achieve this is to first attach dynamic library search
61  /// generators for either the given process, or for the specific required
62  /// libraries, to PlatformJD, then to create the platform instance:
63  ///
64  /// \code{.cpp}
65  /// auto &PlatformJD = ES.createBareJITDylib("stdlib");
66  /// PlatformJD.addGenerator(
67  /// ExitOnErr(EPCDynamicLibrarySearchGenerator
68  /// ::GetForTargetProcess(EPC)));
69  /// ES.setPlatform(
70  /// ExitOnErr(MachOPlatform::Create(ES, ObjLayer, EPC, PlatformJD,
71  /// "/path/to/orc/runtime")));
72  /// \endcode
73  ///
74  /// Alternatively, these symbols could be added to another JITDylib that
75  /// PlatformJD links against.
76  ///
77  /// Clients are also responsible for ensuring that any JIT'd code that
78  /// depends on runtime functions (including any code using TLV or static
79  /// destructors) can reference the runtime symbols. This is usually achieved
80  /// by linking any JITDylibs containing regular code against
81  /// PlatformJD.
82  ///
83  /// By default, MachOPlatform will add the set of aliases returned by the
84  /// standardPlatformAliases function. This includes both required aliases
85  /// (e.g. __cxa_atexit -> __orc_rt_macho_cxa_atexit for static destructor
86  /// support), and optional aliases that provide JIT versions of common
87  /// functions (e.g. dlopen -> __orc_rt_macho_jit_dlopen). Clients can
88  /// override these defaults by passing a non-None value for the
89  /// RuntimeAliases function, in which case the client is responsible for
90  /// setting up all aliases (including the required ones).
91  static Expected<std::unique_ptr<MachOPlatform>>
92  Create(ExecutionSession &ES, ObjectLinkingLayer &ObjLinkingLayer,
93  JITDylib &PlatformJD, const char *OrcRuntimePath,
94  Optional<SymbolAliasMap> RuntimeAliases = None);
95 
96  ExecutionSession &getExecutionSession() const { return ES; }
