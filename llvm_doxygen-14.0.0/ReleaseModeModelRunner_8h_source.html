25 /// ReleaseModeModelRunner - production mode implementation of the
26 /// MLModelRunner. It uses an AOT-compiled SavedModel for efficient execution.
27 template <class TGen>
28 class ReleaseModeModelRunner final : public MLModelRunner {
29 public:
30  /// FeatureNames' type should be an indexed collection of std::string, like
31  /// std::array or std::vector, that has a size() method.
32  template <class FType>
33  ReleaseModeModelRunner(LLVMContext &Ctx, const FType &FeatureNames,
34  StringRef DecisionName, StringRef FeedPrefix = "feed_",
35  StringRef FetchPrefix = "fetch_")
36  : MLModelRunner(Ctx, MLModelRunner::Kind::Release),
37  CompiledModel(std::make_unique<TGen>()) {
38  assert(CompiledModel && "The CompiledModel should be valid");
39 
40  const size_t FeatureCount = FeatureNames.size();
41  FeatureIndices.resize(FeatureCount);
42 
43  for (size_t I = 0; I < FeatureCount; ++I) {
44  const int Index =
45  CompiledModel->LookupArgIndex(FeedPrefix.str() + FeatureNames[I]);
46  assert(Index >= 0 && "Cannot find Feature in inlining model");
47  FeatureIndices[I] = Index;
48  }
49 
50  ResultIndex = CompiledModel->LookupResultIndex(FetchPrefix.str() +
51  DecisionName.str());
52  assert(ResultIndex >= 0 && "Cannot find DecisionName in inlining model");
53  }
54 
55  virtual ~ReleaseModeModelRunner() = default;
56 
57  static bool classof(const MLModelRunner *R) {
58  return R->getKind() == MLModelRunner::Kind::Release;
59  }
60 
61 private:
62  void *evaluateUntyped() override {
63  CompiledModel->Run();
64  return CompiledModel->result_data(ResultIndex);
65  }
66 
67  void *getTensorUntyped(size_t Index) override {
68  return reinterpret_cast<char *>(
69  CompiledModel->arg_data(FeatureIndices[Index]));
70  }
71 
72  std::vector<int32_t> FeatureIndices;
73  int32_t ResultIndex = -1;
74  std::unique_ptr<TGen> CompiledModel;
75 };
