119 const GPUInfo *getArchEntry(AMDGPU::GPUKind AK, ArrayRef<GPUInfo> Table) {
120  GPUInfo Search = { {""}, {""}, AK, AMDGPU::FEATURE_NONE };
121 
122  auto I =
123  llvm::lower_bound(Table, Search, [](const GPUInfo &A, const GPUInfo &B) {
124  return A.Kind < B.Kind;
125  });
126 
127  if (I == Table.end())
128  return nullptr;
129  return I;
130 }

350 // Parse a branch protection specification, which has the form
351 // standard | none | [bti,pac-ret[+b-key,+leaf]*]
352 // Returns true on success, with individual elements of the specification
353 // returned in `PBP`. Returns false in error, with `Err` containing
354 // an erroneous part of the spec.
355 bool ARM::parseBranchProtection(StringRef Spec, ParsedBranchProtection &PBP,
356  StringRef &Err) {
357  PBP = {"none", "a_key", false};
358  if (Spec == "none")
359  return true; // defaults are ok
360 
361  if (Spec == "standard") {
362  PBP.Scope = "non-leaf";
363  PBP.BranchTargetEnforcement = true;
364  return true;
365  }
366 
367  SmallVector<StringRef, 4> Opts;
368  Spec.split(Opts, "+");
369  for (int I = 0, E = Opts.size(); I != E; ++I) {
370  StringRef Opt = Opts[I].trim();
371  if (Opt == "bti") {
372  PBP.BranchTargetEnforcement = true;
373  continue;
374  }
375  if (Opt == "pac-ret") {
376  PBP.Scope = "non-leaf";
377  for (; I + 1 != E; ++I) {
378  StringRef PACOpt = Opts[I + 1].trim();
379  if (PACOpt == "leaf")
380  PBP.Scope = "all";
381  else if (PACOpt == "b-key")
382  PBP.Key = "b_key";
383  else
384  break;
385  }
386  continue;
387  }
388  if (Opt == "")
389  Err = "<empty>";
390  else
391  Err = Opt;
392  return false;
393  }
394 
395  return true;
396 }
